newRegexp :: Text -> Bool -> IO (Either (String, CInt) Regexp)Source

newRegexp regexp case-matters compiles a regular expression pattern regexp into a Regexp value, or returns an error description if the pattern is malformed. The returned CInt is a byte offset into an internally translated pattern, and thus is probably not very useful.
